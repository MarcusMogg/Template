
## 线性筛素数

```cpp
const int ms = 1000010;

bool phi[ms];
int pri[ms], cnt;

void init()
{
    phi[0] = phi[1] = true;
    for (int i = 2; i < ms; ++i)
    {
        if (!phi[i])
        {
            pri[cnt++] = i;
        }
        for (int j = 0; j < cnt&& i*pri[j] < ms; ++j)
        {
            phi[i*pri[j]] = true;
            if (!(i%pri[j]))
            {
                break;
            }
        }
    }
}
```

## 线性筛欧拉函数


```cpp
const int ms = 2e7 + 10;

bool vis[ms];
int pri[ms], cnt, phi[ms];
void init()
{
    vis[0] = vis[1] = true;
    for (int i = 2; i < ms; ++i)
    {
        if (!vis[i])
        {
            pri[cnt++] = i;
            phi[i] = i - 1;
        }
        for (int j = 0; j < cnt&& i*pri[j] < ms; ++j)
        {
            vis[i*pri[j]] = true;
            if (!(i%pri[j]))
            {
                phi[i*pri[j]] = phi[i] * pri[j];
                break;
            }
            phi[i*pri[j]] = phi[i] * (pri[j] - 1);
        }
    }
}
```
